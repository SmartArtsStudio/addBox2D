    /**
     * Abstract base joint class.
     *
     * Joints are used to constraint two bodies together in various fashions.
     * Some joints also feature limits and motors.
     *
     * @class   b2Joint
     * @constructor
     * @param   {b2JointDef}  jointDef
     * @module  Joints
